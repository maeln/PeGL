37 PeDW ObjectManager::load_PeDW(std::string meshfile, std::string imgfile, std::string vshadfile, std::string pshadfile)
38 {
39  PeMesh mesh;
40  PeTexture texture;
41  std::vector<PeShader> shaders;
42  PeProgram program;
43 
44  ShaderLoader shdl;
45  shaders.push_back(shdl.loadShader(vshadfile, GL_VERTEX_SHADER));
46  shaders.push_back(shdl.loadShader(pshadfile, GL_FRAGMENT_SHADER));

48 
49  MeshLoader mshl;

52  ImageLoader imgl;

54 
55  PeDW dw;
56  dw.mesh = mesh;
57  dw.texture = texture;
58  dw.shaders = program;

63 int ObjectManager::draw_PeDW(PeDW obj, glm::vec4 light_position, glm::mat4 world, glm::mat4 perspective, glm::mat4 normal)
64 {
65  glUseProgram(obj.shaders.addr);
66 
67  glBindVertexArray(obj.mesh.vao);
68 
69  glActiveTexture(GL_TEXTURE0 + obj.texture.id);
70  glBindTexture(GL_TEXTURE_2D, obj.texture.addr);
71 
72  // S'occuper des uniforms ici.
73  if ( obj.shaders.uniform.find("world") == obj.shaders.uniform.end()
74  or obj.shaders.uniform.find("perspective") == obj.shaders.uniform.end()
75  or obj.shaders.uniform.find("normalMatrix") == obj.shaders.uniform.end()
76  or obj.shaders.uniform.find("light_dir") == obj.shaders.uniform.end()
77  )
78  {
79  std::cerr << "[ER] Uniform manquant." << std::endl;
80  return 1; // Erreur
81  }
82 
83  glUniformMatrix4fv(obj.shaders.uniform["world"], 1, GL_FALSE, glm::value_ptr(world));
84  glUniformMatrix4fv(obj.shaders.uniform["perspective"], 1, GL_FALSE, glm::value_ptr(perspective));
85  glUniformMatrix3fv(obj.shaders.uniform["normalMatrix"], 1, GL_FALSE, glm::value_ptr(normal));
86  glUniform4fv(obj.shaders.uniform["light_dir"], 1, glm::value_ptr(light_position));
87 
88  glDrawElements(GL_TRIANGLES, obj.mesh.elements_size, GL_UNSIGNED_SHORT, 0);
89 
90  glBindVertexArray(0);
91  glUseProgram(0);
92 
93  return 0; // O -> pas d'erreurs.
94 }
