37 PeShader ShaderLoader::loadShader(std::string filename, GLenum shader_type)
38 {
39  std::string source;
40  PeShader pshader;
41 
42  std::ifstream source_file(filename.c_str());
43  if(!source_file) { std::cerr << "Erreur: Impossible de lire le fichier : " << filename << std::endl; exit(EXIT_FAILURE); }
44  std::string line;
45  while(getline(source_file, line)) // On lit le shader ligne par ligne.
46  {
47  source += (line + '\n');
48  if(line.substr(0, 8) == "uniform ") // Si la ligne commence par «uniform», on note le nom de l'uniform dans un tableau
49  { // pour faciliter l'utilisation des shaders plus tard.
50  std::istringstream str(line.substr(7));
51  std::string type; std::string name;
52  str >> type;
53  str >> name;
54  name.erase(name.find(";"), 1); // La déclaration d'un uniform finit toujours par «;» comme une déclaration en C. Il n'est pas utile de le garder dans le nom de l'uniform.
55  pshader.uniform.push_back(name);
56  }
57  }
58 
59  bool error(false);
60  GLuint tmp(glCreateShader(shader_type));
61  if(tmp == 0) {
62  std::cerr << "Erreur: Problème lors de la création d'un shader : " << filename << std::endl;
63  error = true;
64  }
65 
66  GLint state_s(GL_TRUE);
67  GLint logSize(0);
68  char *log(NULL);
69  const char* src_char = source.c_str();
70  glShaderSource(tmp, 1, &src_char, NULL);
71  glCompileShader(tmp);
72  glGetShaderiv(tmp, GL_COMPILE_STATUS, &state_s);
73  if(state_s != GL_TRUE)
74  {
75  glGetShaderiv(tmp, GL_INFO_LOG_LENGTH, &logSize);
76  log = new char[logSize+1];
77  if(log == NULL)
78  {
79  std::cerr << "Erreur: Impossible d'allouer la mémoire lors de la récupération du Log." << std::endl;
80  error = true;
81  }
82  glGetShaderInfoLog(tmp, logSize, &logSize, log);
83  log[logSize] = '\0';
84  std::cerr << "Erreur: Impossible de compiler le Shader : " << filename << " : " << std::endl << log << std::endl;
85  delete(log);
86  glDeleteShader(tmp);
87  error = true;
88  }
89 
90  if(error == false)
91  {
92  std::cout << "Info: Shader créer avec succès." << std::endl;
93  pshader.addr = tmp;
94  pshader.type = shader_type;
95  }
96 
97  return pshader;
98 }

100 PeProgram ShaderLoader::createProgram(std::vector<PeShader> pshader)
101 {
102  PeProgram oprogram;
103  GLuint tmpProg(glCreateProgram());
104  GLint logSize(0);
105  GLint state_p(GL_TRUE);
106  char *log(NULL);
107  bool error(false);
108 
109  if(pshader.size() > 0)
110  {
111  for(unsigned int i=0; i < pshader.size(); ++i)
112  {
113  glAttachShader(tmpProg, pshader[i].addr);
114  oprogram.shader.push_back(pshader[i].addr);
115  }
116  glLinkProgram(tmpProg);
117  glGetProgramiv(tmpProg, GL_LINK_STATUS, &state_p);
118  if(state_p != GL_TRUE)
119  {
120  glGetProgramiv(tmpProg, GL_INFO_LOG_LENGTH, &logSize);
121  log = new char[logSize+1];
122  if(log == NULL) { std::cerr << "Erreur: Impossible d'allouer la mémoire lors de la récupération du Log." << std::endl; exit(EXIT_FAILURE); }
123  glGetProgramInfoLog(tmpProg, logSize, &logSize, log);
124  log[logSize] = '\0';
125  std::cerr << "Erreur: Impossible de lier le Program : " << log << std::endl;
126  delete(log);
127  glDeleteProgram(tmpProg);
128  error = true;
129  }
130  } else {
131  std::cerr << "Erreur: Pas de shader à lier au program." << std::endl;
132  }
133 
134  if(error == false)
135  {
136  std::cout << "Info: Program lié avec succès." << std::endl;
137  oprogram.addr = tmpProg;
138  for(unsigned int i=0; i < pshader.size(); ++i)
139  {
140  for(unsigned int n=0; n < pshader[i].uniform.size(); ++n)
141  {
142  oprogram.uniform[pshader[i].uniform[n]] = glGetUniformLocation(oprogram.addr, pshader[i].uniform[n].c_str());
143  }
144  }
145  }
146 
147  return oprogram;
148 }
